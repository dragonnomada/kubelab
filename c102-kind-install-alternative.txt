%%% Docker %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

$ sudo snap install docker

$ sudo addgroup --system docker
$ sudo usermod -aG docker $USER
$ sudo snap disable docker
$ sudo snap enable docker

[LOGOUT]

$ docker version
